npm升级到5.0.3之后，安装cli遇到的坑

背景如下：

npm目前升级到了5.0.3

npm install的时候目录会多一个package-lock.json这个文件

此时我在package.json的devDependencies中添加了

"stylus": "^0.54.5",
"stylus-loader": "^3.0.1"

然后在 在终端重新运行npm install的时候，项目中的node_modules并没有出现stylus文件夹

查了下资料说是新版本的坑

https://mp.weixin.qq.com/s?__…
给的解决办法是切换回之前的npm版本……

我的解决过程

1、删除package-lock.json,重新npm install，node_modules中出现了stylus文件夹了，然而还是报错
2、在终端直接运行 sudo npm install stylus-loader stylus --save-dev 结果就编译成功了


在packjson里面设置需要使用的版本号，然后再删除lock文件，再用npm install，即可成功

使用方法，按照文档：
封装jsonp方法：

/**

  • 因为在很多地方都要用到jsonp，所以封装一个通用的方法，传入地址即可调用
    */
    // 导入原始的jsonp库
    import originJSONP from ‘jsonp’

// 暴露一个jsonp方法
// 第二个参数data，通常传给服务端的地址url是带参数的，库是不支持传入一个Objectdata的，需要先把url拼好，再去调用这个库，使用的时候，希望传入一个纯净的url地址，参数用对象的形式传入，拼到url上面，这样调用就更加方便

export default function jsonp (url, data, option) {
    // 判断url参数里面有没有?，如果没有，api/user + '?' + data,r如果有的话，api/user?id=1 + '&' + data
    url += ( url.indexOf('?') < 0 ? '?' : '&') + param (data)
    // jsonp函数返回一个promise对象，有两个参数
    return new Promise((resolve, reject) => {
        // 在promise里面调用原始的originJSONP方法，回调里面，第一个err是null的时候，表示成功
        originJSONP(url, option, (err, data) => {
            if (!err) { // 如果成功，就调用promise的resolve方法
                resolve(data)
            } else {
                reject(err) // 把错误返回出去
            }
        })
    })
}

function param (data) {
    let url = ''
    // 相当于/api/user?id=1&callback=a  ？后面的
    // 遍历传进来的data对象，key:value的形式，通常只有一层，如果多了的话，还需要递归去遍历，并且需要用到hasOwnProperty
    for (var k in data) {
        // 因为不能直接给后端传一个undefined的值，所以需要先判断，没有传递""
        let value = data[k] !== undefined ? data[k] : ''
        url += '&' + k + '=' + encodeURIComponent(value)
        // es6的语法：url += `&${k}=${encodeURIComponent(value)}`
    }
    // 判断url是否存在，如果有，需要把前面的&去掉,如果没有，返回空
    return url ? url.substring(1) : ''
}

手动代理。

在dev-server.js里面设置

// 设置代理请求，把前端跨域的地址放在后端
var apiRoutes = express.Router()
apiRoutes.get('/getDiscList', function(req, res) {
var url = 'https://c.y.qq.com/splcloud/fcgi-bin/fcg_get_diss_by_tag.fcg'
axios.get(url, {
    headers: {  // 重点，修改为qq的请求接口数据
      referer: 'https://c.y.qq.com/',
      host: 'c.y.qq.com'
    },
    params: req.query // 前端传递过来的参数
  }).then((response) => {
    res.json(response.data) // 把qq返回来的数据传递给node返回给前台的数据
  }).catch((e) => {
    console.log(e)
  })


})

vuex

从歌手列表页切入到歌手详情页，需要把歌手的id，name.图片都传入，可以通过组件之间传值的方式，event-bus，再新建一个空白vue实例，作为中间件，也可以通过父组件作为桥梁，但是参数比较多的时候，是非常麻烦的，就引入了vuex

vuex:

1.是什么？状态管理模式，可以管理所有组件的所有状态，把这些状态的数据放在一个内存空间中去管理，称为state，state里面的数据，可以方便的映射到组件上，渲染组件。当组件的数据发生变化的时候，可以通过dispatch一个action，这个action可以去做一个一个异步操作，比如以后的那些交互。

然后action去commit一个notation,也可以直接在组件里面去commit一个notation，这个notation是一个唯一可以修改state的途径，其他任何方式去修改都是非法的，目的就是让state状态修改可以预测。

state的修改又会反应到组件上，这样就实现了一个闭环。

2、在vue里面，是可以直接修改组件的数据的，数据的修改会因映射到组件上，必须通过dispatch一个action或直接commit一个notation来修改state里的数据，反而变得繁琐。

3、使用场景，解决的问题：
1) 来解决多个组件之间的状态共享，假设应用比较复杂，其中一些数据是被组件共享的，而这些组件又是兄弟组件，甚至是关联度很低的组件，这时想要共享数据就比较困难。包括去修改数据，组件发生变化，都会非常困难，通过eventbus或其他方式，很复杂，也不好维护；

2）来解决路由间的复杂传递，遇到一些路由跳转场景，需要传递的参数很复杂的时候，用vuex传递数据很方便
如果是小型应用的话，不需要使用，大材小用。

autoperfix方法封装

 利用浏览器能力监测方法
// 封装prefixed方法,会根据浏览器的支持能力去自动确定添加前缀
 let elementStyle = document.createElement('div').style

 let vendor = (() => {
    // 列举浏览器支持的
    let transformNames = {
        webkit: 'webkitTransform',
        Moz: 'MazTransform',
        O: 'OTransform',
        ms: 'msTransform',
        standard: 'transform'
    }
    // 测试浏览器支持哪一种
    for (let key in transformNames) {
        if (elementStyle[transformNames[key]] !== undefined) {
            return key
        }
    }

    // 如果所有的都不支持，则浏览器有故障
    return false
 })()

 export function prefixStyle (style) {
    if (vendor === false) {
        return false
    }
    // 支持标准的
    if (vendor === 'standard') {
        return style
    }

    return vendor + style.charAt(0).toUpperCase() + style.substr(1)


 }

布局方面

如果需要左边图片固定宽度，右边自适应，可以用flex布局

父元素： display: flxe
左边：flex: 0 0 40px 
           width： 40px
          (第一个0是指flex-grow,定义项目的放大比例，默认为0，有剩余空间的时候，别的放大，它也不放大；第二个0是flex-shrink属性，空间不够，别的项目都缩小的时候，它也不缩小，还是固定的；第三个值是flex-basis属性，定义了在分配多余空间之前，自己的宽度高度，默认值是auto，可以给一个固定的宽度)
右边： flex: 1 // 随着浏览器的宽度自适应，方法或缩小
